async function enviarMensaje(){ const nombre=document.getElementById("nombre").value; const empresa=document.getElementById("empresa").value; const telefono=document.getElementById("telefono").value; const mensaje=document.getElementById("mensaje").value; if(nombre=="" || mensaje==""){ alert("Ingrese al menos su nombre y un mensaje."); return; } const respuesta=document.getElementById("respuesta"); respuesta.innerHTML="⏳ Enviando..."; try{ const r=await fetch("https://smartdoor-vistony.vistonyccp.workers.dev/",{ method:"POST", headers:{ "Content-Type":"application/json" }, body:JSON.stringify({ mensaje: `📩 NUEVO MENSAJE 👤 ${nombre} 🏢 ${empresa} 📞 ${telefono} 💬 ${mensaje}` }) }); if(r.ok){ respuesta.innerHTML="✅ Mensaje enviado correctamente."; document.getElementById("nombre").value=""; document.getElementById("empresa").value=""; document.getElementById("telefono").value=""; document.getElementById("mensaje").value=""; }else{ respuesta.innerHTML="❌ No fue posible enviar el mensaje."; } }catch(e){ respuesta.innerHTML="❌ Error de conexión."; } }
